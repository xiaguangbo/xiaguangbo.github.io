## 简介

自动构建系统，使用 C# + .NET 7 开发

会自动记录构建过程，有任何未完成的情况都会在其基础上继续运行

### 启动构建

```
cd rockchip_ubuild
dotnet ubuild/main.cs rockchip/configuration/firefly_rk3399.ub.md
```

### .ub.md 的语法

```
## value

<!-- 变量，值为字符串 -->

tool_path = "../../../tool/"

## rely

<!-- 执行这些文件并等待这些文件执行结束 -->

tool_path + "git/git.ub.md"
tool_path + "make/make.ub.md"

## download

<!-- 会进入 build/download/此文件名（没有 .ub.md 后缀） 执行命令 -->

"git --help"
"cp --help"

## source

<!-- 进入 build/source/此文件名 里执行命令 -->
<!-- 需要复制一份或解压下载的压缩包到这里 -->

"cp --help"
"tar --help"

## patch

<!-- 进入 build/source/此文件名 里执行命令 -->

"patch --help"

## build

<!-- 会进入 build/source/此文件名 里执行命令 -->

"make --help"

## image

<!-- 会进入 build/image 执行命令 -->

"cp --help"
```

## 示例文件的内容

### firefly_rk3399.ub.md

```
## rely

"../bootload/u-boot/u-boot.ub.md"
"../kernel/linux/linux.ub.md"
"../rootfs/busybox/busybox.ub.md"

## image

"
cp ../source/u-boot/idbloader.img ./
cp ../source/u-boot/u-boot.itb ./"
cp ../source/linux/boot.img ./"
cp ../source/busybox/rootfs.img ./
"

"
dd if=/dev/zero of=linux.img bs=1M count=1024

fdisk linux.img << EOF
g
n

32768
+64M
n

262144

w
EOF

dd if=idbloader.img of=linux.img bs=1k seek=32 conv=notrunc
dd if=u-boot.itb of=linux.img bs=1M seek=8 conv=notrunc
dd if=boot.img of=linux.img bs=1M seek=16 conv=notrunc
dd if=rootfs.img of=linux.img bs=1M seek=128
"
```

### arm-trusted-firmware.ub.md

```
## download

git clone --depth=1 https://github.com/ARM-software/arm-trusted-firmware.git"

## source

"cp ../download/arm-trusted-firmware/arm-trusted-firmware ./"

## build

"
cd arm-trusted-firmware
make PLAT=rk3399 CROSS_COMPILE=aarch64-linux-gnu-
"
```

### u-boot.ub.md

```
## rely

"../arm-trusted-firmware/arm-trusted-firmware.ub.md"

## download

"git clone --depth=1 https://github.com/u-boot/u-boot.git"

## source

"cp ../download/u-boot/u-boot ./"

## build

"
cd u-boot
cp ../../arm-trusted-firmware/build/rk3399/release/bl31/bl31.elf ./

"
```

### linux.ub.md

```
## download

"git clone --depth=1 https://github.com/torvalds/linux.git"

## source

"cp ../download/linux/linux ./"

## build

"cd linux"
"make ARCH=arm64 CROSS_COMPILE=aarch64-linux-gnu- defconfig"
"make ARCH=arm64 CROSS_COMPILE=aarch64-linux-gnu- -j$(nproc)"

"
mkdir boot_img
cp arch/arm64/boot/dts/rockchip/rk3399-firefly.dtb boot_img/
cp arch/arm64/boot/Image boot_img/
mkdir boot_img/extlinux
cat>boot_img/extlinux/extlinux.conf<<EOF
label rockchip-kernel-5.19.2
    kernel /Image
    fdt /rk3399-firefly.dtb
    append earlycon=uart8250,mmio32,0xff1a0000 root=/dev/mmcblk1p2 rootwait rootfstype=ext4
EOF
"

"
rm -rf boot_img_mount
dd if=/dev/zero of=boot.img bs=1M count=64
mkfs.fat boot.img
mkdir boot_img_mount
mount boot.img boot_img_mount/
cp -r boot_img/* boot_img_mount/
umount boot_img_mount
"
```

### busybox.ub.md

```
## download

"git clone --depth=1 https://github.com/mirror/busybox.git"

## source

"cp ../download/busybox/busybox ./"

## patch

"patch --help"

## build

"
cd busybox
make
make install
"

"
cd _install
mkdir etc dev mnt
mkdir -p proc sys tmp mnt
mkdir -p etc/init.d/
cat>etc/fstab<<EOF
proc /proc proc defaults 0 0
tmpfs /tmp tmpfs defaults 0 0
sysfs /sys sysfs defaults 0 0
EOF

cat>etc/init.d/rcS<<EOF
echo -e "Welcome to Linux"
/bin/mount -a
echo -e "Remounting the root filesystem"
mount  -o  remount,rw  /
mkdir -p /dev/pts
mount -t devpts devpts /dev/pts
echo /sbin/mdev > /proc/sys/kernel/hotplug
mdev -s
EOF

chmod 755 etc/init.d/rcS
cat>etc/inittab<<EOF
::sysinit:/etc/init.d/rcS
::respawn:-/bin/sh
::askfirst:-/bin/sh
::cttlaltdel:/bin/umount -a -r
EOF

chmod 755 etc/inittab
cd dev
mknod console c 5 1
mknod null c 1 3
mknod tty1 c 4 1
"

"
cd ..
rm -rf rootfs_img_mount
dd if=/dev/zero of=rootfs.img bs=1M count=32
mkfs.ext4 rootfs.img
mkdir rootfs_img_mount
mount -o loop rootfs.img ./rootfs_img_mount
cp -rf ./_install/* ./rootfs_img_mount
umount ./rootfs_img_mount
"
```
